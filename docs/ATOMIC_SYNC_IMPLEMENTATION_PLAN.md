# Atomic Sync Implementation Plan (Normalized)

**Status**: Planning Phase  
**Target**: Complete replacement of existing batch sync with atomic grouped operations  
**Timeline**: 8 weeks (no backward compatibility needed)

---

## Overview

Replace the current entity-by-entity sync with atomic group sync where related entities (expense + tags + relationships) are synced as a single transaction. Either all succeed or all fail together.

**Key Principles**:
- ✅ Normalized database design for sync groups
- ✅ True ACID guarantees per group
- ✅ Complete API replacement (breaking change)
- ✅ Foreign key integrity enforced
- ✅ Simplified architecture

```python
from typing import List, Dict
from sqlalchemy.orm import Session
from sqlalchemy import exc
import logging

logger = logging.getLogger(__name__)


class AtomicSyncService:
    """
    Processes atomic sync groups with ACID guarantees.
    Each group is processed in a savepoint (nested transaction).
    """
    
    def __init__(self, db: Session):
        self.db = db
        self.entity_mappings: Dict[str, str] = {}  # clientId -> serverId
        
    def process_groups(
        self,
        groups: List[AtomicSyncGroup],
        client_timestamp: int
    ) -> List[AtomicGroupResult]:
        """Process all groups. Each group is independent transaction."""
        results = []
        
        for group in groups:
            result = self.process_single_group(group, client_timestamp)
            results.append(result)
            
        return results
    
    def process_single_group(
        self,
        group: AtomicSyncGroup,
        client_timestamp: int
    ) -> AtomicGroupResult:
        """
        Process one atomic group in a savepoint.
        All operations succeed or all rollback.
        """
        # Start savepoint for this group
        savepoint = self.db.begin_nested()
        
        try:
            entity_mappings = []
            
            # Process operations in sequence
            for operation in group.operations:
                mapping = self._process_operation(operation, group.group_id)
                if mapping:
                    entity_mappings.append(mapping)
            
            # All succeeded - commit savepoint
            savepoint.commit()
            
            # Log success
            self._log_sync_audit(
                group_id=group.group_id,
                group_type=group.group_type,
                status="success",
                entity_count=len(group.operations),
                client_timestamp=client_timestamp
            )
            
            return AtomicGroupResult(
                group_id=group.group_id,
                success=True,
                entity_mappings=entity_mappings,
                rolled_back=False
            )
            
        except Exception as e:
            # Rollback this group's savepoint
            savepoint.rollback()
            
            logger.error(
                f"Group {group.group_id} failed: {str(e)}",
                exc_info=True
            )
            
            # Log failure
            self._log_sync_audit(
                group_id=group.group_id,
                group_type=group.group_type,
                status="failed",
                entity_count=len(group.operations),
                client_timestamp=client_timestamp,
                error_message=str(e)
            )
            
            return AtomicGroupResult(
                group_id=group.group_id,
                success=False,
                error=str(e),
                entity_mappings=[],
                rolled_back=True
            )
    
    def _process_operation(
        self,
        operation: SyncOperation,
        group_id: str
    ) -> Optional[EntityMapping]:
        """Process a single operation within a group."""
        
        if isinstance(operation, CreateExpenseBatchRequest):
            return self._create_expense(operation, group_id)
        elif isinstance(operation, CreateTagBatchRequest):
            return self._create_tag(operation, group_id)
        elif isinstance(operation, CreateExpenseTagBatchRequest):
            return self._create_expense_tag(operation, group_id)
        # ... handle all operation types
        
    def _create_expense(
        self,
        operation: CreateExpenseBatchRequest,
        group_id: str
    ) -> EntityMapping:
        """Create expense and return client->server ID mapping."""
        
        new_expense = Expense(
            title=operation.title,
            amount=operation.amount,
            year=operation.year,
            month=operation.month,
            date=operation.date,
            created_at=datetime.utcnow(),
            updated_at=datetime.utcnow()
        )
        
        self.db.add(new_expense)
        self.db.flush()  # Get ID without committing
        
        # Track mapping for use in same group
        self.entity_mappings[operation.client_id] = new_expense.id
        
        return EntityMapping(
            entity_type="expense",
            client_id=operation.client_id,
            server_id=str(new_expense.id)
        )
    
    def _create_expense_tag(
        self,
        operation: CreateExpenseTagBatchRequest,
        group_id: str
    ) -> EntityMapping:
        """
        Create expense-tag relationship.
        Resolves client IDs to server IDs using mappings from same group.
        """
        
        # Resolve expense ID (might be from earlier in this group)
        expense_id = self._resolve_id(operation.expense_id, "expense")
        tag_id = self._resolve_id(operation.tag_id, "tag")
        
        new_expense_tag = ExpenseTagsCrossRef(
            expense_id=expense_id,
            tag_id=tag_id,
            created_at=datetime.utcnow(),
            updated_at=datetime.utcnow()
        )
        
        self.db.add(new_expense_tag)
        self.db.flush()
        
        return EntityMapping(
            entity_type="expense_tag",
            client_id=operation.client_id,
            server_id=str(new_expense_tag.id)
        )
    
    def _resolve_id(self, client_id: str, entity_type: str) -> str:
        """
        Resolve client ID to server ID.
        First checks current group mappings, then database.
        """
        # Check if we created this in current group
        if client_id in self.entity_mappings:
            return self.entity_mappings[client_id]
        
        # Query database for existing mapping
        # (from previous sync)
        # Implementation depends on your entity structure
        # Could query the entity table directly by some local_id field
        
        # If not found, assume it's already a server ID
        return client_id
    
    def _log_sync_audit(
        self,
        group_id: str,
        group_type: str,
        status: str,
        entity_count: int,
        client_timestamp: int,
        error_message: Optional[str] = None
    ):
        """Log sync attempt for monitoring."""
        
        audit = SyncAuditLog(
            group_id=group_id,
            group_type=group_type,
            client_timestamp=client_timestamp,
            status=status,
            error_message=error_message,
            entity_count=entity_count
        )
        self.db.add(audit)
```

```python
from sqlalchemy.orm import Session

from ..database import get_db

router = APIRouter()


@router.post("/atomic", response_model=AtomicSyncResponse)
async def atomic_sync(
    request: AtomicSyncRequest,
    db: Session = Depends(get_db)
):
    """
    Atomic batch sync endpoint.
    Each group is processed as an atomic transaction.
    """
    try:
        service = AtomicSyncService(db)
        
        # Process all groups
        group_results = service.process_groups(
            groups=request.groups,
            client_timestamp=request.client_timestamp
        )
        
        # Commit outer transaction (all savepoints already committed/rolled back)
        db.commit()
        
        return AtomicSyncResponse(
            group_results=group_results,
            server_timestamp=int(time.time() * 1000)
        )
        
    except Exception as e:
        db.rollback()
        raise HTTPException(status_code=500, detail=str(e))
```

```kotlin
class SyncManager(
    private val database: AppDatabase,
    private val apiService: FinanceHubApiService,
    private val syncGroupDao: SyncGroupDao,
    private val entityMappingDao: EntityMappingDao
) {
    
    suspend fun performFullSync(): SyncResult {
        try {
            _syncState.value = SyncState.SYNCING
            
            // Push atomic groups
            val pushResult = pushAtomicGroups()
            if (pushResult is SyncResult.Failure) {
                _syncState.value = SyncState.ERROR
                return pushResult
            }
            
            // Pull server changes
            val pullResult = pullServerChanges()
            if (pullResult is SyncResult.Failure) {
                _syncState.value = SyncState.ERROR
                return pullResult
            }
            
            _syncState.value = SyncState.IDLE
            return SyncResult.success("Sync completed")
            
        } catch (e: Exception) {
            _syncState.value = SyncState.ERROR
            return SyncResult.failure("Sync failed: ${e.message}")
        }
    }
    
    private suspend fun pushAtomicGroups(): SyncResult {
        try {
            // Get all pending groups
            val pendingGroups = syncGroupDao.getPendingSyncGroups()
            
            if (pendingGroups.isEmpty()) {
                return SyncResult.success("No pending groups")
            }
            
            // Build request
            val groupOperations = pendingGroups.map { group ->
                buildAtomicSyncGroup(group)
            }
            
            val request = AtomicSyncRequest(
                groups = groupOperations,
                clientTimestamp = System.currentTimeMillis()
            )
            
            // Send to server
            val response = apiService.atomicSync(request)
            
            if (response.isSuccessful) {
                val responseBody = response.body()!!
                
                // Process each group result
                responseBody.groupResults.forEach { result ->
                    processGroupResult(result)
                }
                
                return SyncResult.success("Pushed ${pendingGroups.size} groups")
            } else {
                return SyncResult.failure("Sync failed: ${response.message()}")
            }
            
        } catch (e: Exception) {
            Log.e(TAG, "Failed to push atomic groups", e)
            return SyncResult.failure("Push failed: ${e.message}")
        }
    }
    
    private suspend fun buildAtomicSyncGroup(group: SyncGroup): AtomicSyncGroupRequest {
        // Get all entities in group
        val entities = syncGroupDao.getGroupEntities(group.groupId)
        
        // Convert to operations
        val operations = entities.map { entity ->
            buildOperation(entity)
        }
        
        return AtomicSyncGroupRequest(
            groupId = group.groupId,
            groupType = group.groupType,
            operations = operations
        )
    }
    
    private suspend fun buildOperation(entity: SyncGroupEntity): Any {
        // Based on entity type and operation, fetch data and create request
        return when (entity.entityType) {
            "expense" -> {
                val expense = database.expenseDao().getExpenseById(entity.entityId.toInt())!!
                CreateExpenseBatchRequest(
                    title = expense.title,
                    amount = expense.amount,
                    year = expense.year,
                    month = expense.month,
                    date = expense.date,
                    clientId = entity.entityId
                )
            }
            "tag" -> {
                val tag = database.tagsDao().getTagById(entity.entityId.toInt())!!
                if (entity.operation == "CREATE") {
                    CreateTagBatchRequest(
                        name = tag.tag,
                        monthlyAmount = tag.monthlyAmount,
                        currentMonth = tag.currentMonth,
                        currentYear = tag.currentYear,
                        createdDay = tag.createdDay,
                        createdMonth = tag.createdMonth,
                        createdYear = tag.createdYear,
                        clientId = entity.entityId
                    )
                } else {
                    UpdateTagBatchRequest(
                        serverId = entity.serverId!!,
                        name = tag.tag,
                        monthlyAmount = tag.monthlyAmount,
                        currentMonth = tag.currentMonth,
                        currentYear = tag.currentYear
                    )
                }
            }
            "expense_tag" -> {
                val parts = entity.entityId.split("-")
                val expenseId = resolveEntityId("expense", parts[0])
                val tagId = resolveEntityId("tag", parts[1])
                
                CreateExpenseTagBatchRequest(
                    expenseId = expenseId,
                    tagId = tagId,
                    clientId = entity.entityId
                )
            }
            else -> throw IllegalArgumentException("Unknown entity type: ${entity.entityType}")
        }
    }
    
    private suspend fun resolveEntityId(entityType: String, localId: String): String {
        // Check if we have a server mapping
        val serverId = entityMappingDao.getServerId(entityType, localId)
        return serverId ?: localId  // Use local ID as fallback (for same-group references)
    }
    
    private suspend fun processGroupResult(result: AtomicGroupResult) {
        if (result.success) {
            // Mark group as synced
            syncGroupDao.updateGroupStatus(
                groupId = result.groupId,
                status = "synced",
                syncedAt = System.currentTimeMillis()
            )
            
            // Store entity mappings
            val mappings = result.entityMappings.map { mapping ->
                EntityMapping(
                    entityType = mapping.entityType,
                    localId = mapping.clientId,
                    serverId = mapping.serverId,
                    syncedAt = System.currentTimeMillis()
                )
            }
            entityMappingDao.insertMappings(mappings)
            
            // Update entity serverIds
            result.entityMappings.forEach { mapping ->
                updateEntityServerId(mapping)
            }
            
        } else {
            // Mark group as failed
            syncGroupDao.markGroupFailed(
                groupId = result.groupId,
                status = "failed",
                error = result.error ?: "Unknown error",
                timestamp = System.currentTimeMillis()
            )
        }
    }
    
    private suspend fun updateEntityServerId(mapping: EntityMapping) {
        when (mapping.entityType) {
            "expense" -> database.expenseDao().updateServerId(
                mapping.clientId.toInt(),
                mapping.serverId
            )
            "tag" -> database.tagsDao().updateServerId(
                mapping.clientId.toInt(),
                mapping.serverId
            )
            // ... other entity types
        }
    }
}
```
